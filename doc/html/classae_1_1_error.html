Handle errors.

Example :

class MyError : public ae::Error {
MyError(int code);
virtual MyError();
}
MyError::MyError(int code) {
if(code < 0)
flags_.set_error();
}
int main() {
MyError error(-1);
if (error.IsError) throw error.Message();
}
